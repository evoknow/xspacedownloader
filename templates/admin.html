        python admin_ads.py list

        python admin_ads.py create --copy "Your ad HTML" --start "2025-01-01 00:00:00" --end "2025-12-31 23:59:59" --max-impressions 1000

        python admin_ads.py activate 3

        python admin_ads.py suspend 3
